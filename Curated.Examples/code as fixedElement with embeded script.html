public static void Eggplant()
{
    int a,b,sum,diff;
    for (int i=1; i<=4; i++)
    {
        a = int.Parse(Console.ReadLine());
        b = int.Parse(Console.ReadLine());
        sum=a+b;
        diff=b-a;
        if ( sum<=diff )
            Console.Write("Hello ");
        else
            Console.Write("Bye ");
    }
}
